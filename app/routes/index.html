API

Convert any datamodel and save to database without any DML/DDL.

API USAGE



#INSTALLATION#
npm init
npm i express sequelize sequelize-cli pg mysql mysql2 validator body-parser dotenv --save


#USAGE OPTION1
    #ToSend data to database with any json
    /push send body as example below
    EXAMPLE1: Send data with empid as unique, then the data at that particular record of empid is deleted and below updated
            {
        	"tableName": "teachers",
        	"unique": "empid",
        	"items": [{
        			"name": "san",
                    "empid": "1",
        			"place": "hsn",
        			"city":"dubai",
        			"salary":12677
        		},
        		{
        			"name": "sud",
                    "empid": "2",
        			"place": "mys",
        			"city":"abu",
        			"salary":9999
        		}
        	]
        }

    EXAMPLE2: Send data with data to be replaced at ["testname","weeknumber"], then the data at that particular record of testname && weeknumber is replaced
            {
        	"tableName": "testcases",
        	"unique": ["testname","weeknumber"]
        	"items": [{
        			"testname": "tqc1",
                    "weeknumber": 10,
                    "status": "passed"
        			"system":"codex"
        		},
                {
        			"testname": "tqc2",
                    "weeknumber": 10,
                    "status": "passed"
        			"system":"codex"
        		}
        	]
        }
